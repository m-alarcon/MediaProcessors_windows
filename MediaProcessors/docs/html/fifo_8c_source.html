183 void fifo_close(fifo_ctx_t **pfifo_ctx)
184 {
185  fifo_ctx_t *fifo_ctx;
186 
187  if(pfifo_ctx== NULL)
188  return;
189 
190  if((fifo_ctx= *pfifo_ctx)!= NULL) {
191 
192  /* Set exit flag and send signals to eventually unlock mutex */
193  fifo_ctx->flag_exit= 1;
194  pthread_mutex_lock(&fifo_ctx->api_mutex);
195  pthread_cond_broadcast(&fifo_ctx->buf_put_signal);
196  pthread_cond_broadcast(&fifo_ctx->buf_get_signal);
197  pthread_mutex_unlock(&fifo_ctx->api_mutex);
198 
199  /* Release FIFO buffer */
200  if(fifo_ctx->buf!= NULL) {
201  /* Check and release each element of buffer */
202  int i;
203  for(i= 0; i< fifo_ctx->buf_max_size; i++) {
204  fifo_elem_ctx_t *fifo_elem_ctx= &fifo_ctx->buf[i];
205  if(fifo_elem_ctx->elem!= NULL) {
206  if(fifo_ctx->elem_ctx_release!= NULL) {
207  fifo_ctx->elem_ctx_release(&fifo_elem_ctx->elem);
208  } else {
209  free(fifo_elem_ctx->elem);
210  }
211  fifo_elem_ctx->elem= NULL;
212  fifo_elem_ctx->size= 0;
213  }
214  }
215  free(fifo_ctx->buf);
216  fifo_ctx->buf= NULL;
217  }
218 
219  /* Release mutex and conditionals */
220  pthread_mutex_destroy(&fifo_ctx->api_mutex);
221  pthread_cond_destroy(&fifo_ctx->buf_put_signal);
222  pthread_cond_destroy(&fifo_ctx->buf_get_signal);
223 
224  /* Release module instance context structure */
225  free(fifo_ctx);
226  *pfifo_ctx= NULL;
227  }

266 int fifo_traverse(fifo_ctx_t *fifo_ctx, int elem_cnt,
267  void (*it_fxn)(void *elem, ssize_t elem_size, int idx, void *it_arg,
268  int *ref_flag_break),
269  void *it_arg)
270 {
271  ssize_t buf_level;
272  int i, cnt, cnt_max, flag_break;
273  uint32_t buf_max_size;
274  LOG_CTX_INIT(NULL);
275 
276  /* Check arguments */
277  CHECK_DO(fifo_ctx!= NULL, return STAT_ERROR);
278  CHECK_DO(elem_cnt> 0 || elem_cnt== -1, return STAT_ERROR);
279  CHECK_DO(it_fxn!= NULL, return STAT_ERROR);
280 
281  /* Lock API MUTEX */
282  pthread_mutex_lock(&fifo_ctx->api_mutex);
283 
284  /* Iterate: we do it beginning with the input index (namely, we go from
285  * the newest queued element to the oldest).
286  */
287  buf_level= fifo_ctx->buf_level;
288  if(elem_cnt== -1)
289  elem_cnt= buf_level; // '-1' means "traverse all the FIFO"
290  cnt_max= (elem_cnt< buf_level)? elem_cnt: buf_level;
291  buf_max_size= fifo_ctx->buf_max_size;
292  flag_break= 0;
293  for(i= fifo_ctx->input_idx- 1, cnt= 0; cnt< cnt_max; cnt++) {
294  fifo_elem_ctx_t fifo_elem_ctx= fifo_ctx->buf[i];
295 
296  /* Execute iterator callback function */
297  it_fxn(fifo_elem_ctx.elem, fifo_elem_ctx.size, i, it_arg, &flag_break);
298  if(flag_break!= 0)
299  break;
300 
301  /* Update for next iteration; note that 'buf_max_size' is > 0 in
302  * modulo operation:
303  * integer r = a % b;
304  * r= r < 0 ? r + b : r; <- Only works if B> 0
305  */
306  i= (i- 1)% buf_max_size;
307  if(i< 0)
308  i= i+ buf_max_size;
309  }
310 
311  pthread_mutex_unlock(&fifo_ctx->api_mutex);
312  return STAT_SUCCESS;
313 }

347 static inline int fifo_input(fifo_ctx_t *fifo_ctx, void **ref_elem,
348  size_t elem_size, int dup_flag)
349 {
350  uint32_t buf_max_size;
351  fifo_elem_ctx_t *fifo_elem_ctx= NULL;
352  int end_code= STAT_ERROR;
353  LOG_CTX_INIT(NULL);
354 
355  /* Check arguments */
356  CHECK_DO(fifo_ctx!= NULL, return STAT_ERROR);
357  CHECK_DO(ref_elem!= NULL && *ref_elem!= NULL, return STAT_ERROR);
358  CHECK_DO(elem_size> 0, return STAT_ERROR);
359 
360  /* Lock API mutex */
361  pthread_mutex_lock(&fifo_ctx->api_mutex);
362 
363  buf_max_size= fifo_ctx->buf_max_size;
364 
365  /* Block until we have space in buffer to put new element */
366  while(fifo_ctx->buf_level>= buf_max_size &&
367  !(fifo_ctx->flags& FIFO_O_NONBLOCK) &&
368  fifo_ctx->flag_exit== 0) {
369  pthread_cond_broadcast(&fifo_ctx->buf_put_signal);
370  pthread_cond_wait(&fifo_ctx->buf_get_signal, &fifo_ctx->api_mutex);
371  }
372  /* If 'FIFO_O_NONBLOCK' flag is set, check if buffer has available space
373  * before proceeding.
374  */
375  if(fifo_ctx->buf_level>= buf_max_size &&
376  (fifo_ctx->flags& FIFO_O_NONBLOCK)) {
377  //LOGV("FIFO buffer overflow!\n"); //Comment-me
378  end_code= STAT_ENOMEM;
379  goto end;
380  }
381 
382  /* Allocate, copy and insert element */
383 
384 
385  /* Get FIFO slot where to put new element */
386  fifo_elem_ctx= &fifo_ctx->buf[fifo_ctx->input_idx];
387  CHECK_DO(fifo_elem_ctx->elem== NULL, goto end);
388  CHECK_DO(fifo_elem_ctx->size== 0, goto end);
389 
390  /* Get or copy (if applicable) the new element */
391  if(dup_flag!= 0 && fifo_ctx->elem_ctx_dup!= NULL) {
392  fifo_elem_ctx->elem= fifo_ctx->elem_ctx_dup(*ref_elem);
393  CHECK_DO(fifo_elem_ctx->elem!= NULL, goto end);
394  }
395  if(dup_flag!= 0 && fifo_ctx->elem_ctx_dup== NULL) {
396  fifo_elem_ctx->elem= malloc(elem_size);
397  CHECK_DO(fifo_elem_ctx->elem!= NULL, goto end);
398  memcpy(fifo_elem_ctx->elem, *ref_elem, elem_size);
399  }
400  if(dup_flag== 0) {
401  fifo_elem_ctx->elem= *ref_elem;
402  *ref_elem= NULL; // we now own the element
403  }
404  fifo_elem_ctx->size= elem_size;
405 
406  /* Update circular buffer management variables */
407  fifo_ctx->buf_level+= 1;
408  //CHECK_DO(fifo_ctx->buf_level<= buf_max_size,
409  // fifo_ctx->buf_level= buf_max_size); //comment-me
410  fifo_ctx->input_idx= (fifo_ctx->input_idx+ 1)% buf_max_size;
411  pthread_cond_broadcast(&fifo_ctx->buf_put_signal);
412 
413  end_code= STAT_SUCCESS;
414 end:
415  pthread_mutex_unlock(&fifo_ctx->api_mutex);
416  return end_code;
417 }

444  //LOGV("FIFO buffer underun!\n"); //comment-me
445  pthread_cond_broadcast(&fifo_ctx->buf_get_signal);
446  pthread_cond_wait(&fifo_ctx->buf_put_signal, &fifo_ctx->api_mutex);

449  * proceeding.
450  */

452  //LOGV("FIFO buffer underun!\n"); //comment-me
453  end_code= STAT_EAGAIN;

455  }

464  fifo_elem_ctx->elem= NULL;
465  fifo_elem_ctx->size= 0;
466

473  }
474 
475  end_code= STAT_SUCCESS;
476 end:
477  pthread_mutex_unlock(&fifo_ctx->api_mutex);
478  return end_code;
